import codecademylib
import pandas as pd

visits = pd.read_csv('visits.csv',
                     parse_dates=[1])
cart = pd.read_csv('cart.csv',
                   parse_dates=[1])

                       parse_dates=[1])

                       parse_dates=[1])
print(visits.head())
print(cart.head())
print(checkout.head())
print(purchase.head())

print visits_cart
l1 = len(visits_cart)

null_rows = visits_cart[visits_cart.cart_time.isnull()]
print null_rows

percent_nocart = float(len(null_rows))/(l1+len(null_rows))
print percent_nocart

print cart_checkout
